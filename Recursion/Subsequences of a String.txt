

#include <iostream>
#include <string.h>
#include <math.h>

using namespace std;

int Subsequence(string input,string output[])
{
    if(input.empty())
    {
        output[0]="";
        return 1;
    }
    else
    {
        string Samllstring=input.substr(1);
        int smallOuput_size=Subsequence(Samllstring,output);
        for(int i=0;i<smallOuput_size;i++)
        {
            output[i+smallOuput_size]=input[0]+output[i];
        }
        return 2 * smallOuput_size;
    }
}

int main()
{
    string input;
    cin>>input;
    
    int size=input.length();
    int output_size=pow(2,size);
    
    string *output=new string[output_size];
    int count=Subsequence(input,output);
    for(int i=0;i<count;i++)
    {
        cout<<output[i]<<'\n';
    }
    
    return 0;
}
